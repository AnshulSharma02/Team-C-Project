	agent any
	stages{

			steps{
				bat 'javac src\\SimpleCalculator.java'

				}
		}
		stage('TestBuild'){
			steps{
				bat 'javac -cp junit-4.12.jar SimpleCalculator.java SimpleCalculatorTests.java'
				bat 'java -cp junit-4.12.jar;hamcrest-core-1.3.jar;. org.junit.runner.JUnitCore SimpleCalculatorTests'
				}
		}
	}	
}
